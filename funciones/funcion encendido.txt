void encendido(){
digitalWrite(pinLatch,LOW);
int encendido[2]={};

for(int i=0;i<1;i++){
	encendido[i]=1;
}

digitalWrite(pinData, encendido[0]);    
for(int i=0;i<8;i++){
  for(int j=0;j<8;j++){
    digitalWrite(pinClock,HIGH);
    digitalWrite(pinClock, LOW);
  	}
  
}
  digitalWrite(pinLatch,HIGH);
}
